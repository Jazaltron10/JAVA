bool DateClass::validate()
{

	bool ok;
	if ((day >= 1 && day <= 28 && month == 2)
		|| (day >= 1 && day <= 30 && (month == 9 || month == 4 || 
			month == 6 || month == 11)) 
		|| (day >= 1 && day <= 31 && (month == 1 || month == 3 || 
			month == 5 || month == 7 || 
			month == 8 || month == 10 || month == 12)))
	{
		ok = true;
		return ok;
	}
	else {
		ok = false;
		return ok;
	}
}
